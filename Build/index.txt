  <body>
    <div id="unity-container">
      <canvas id="unity-canvas" width={{{ WIDTH }}} height={{{ HEIGHT }}} tabindex="-1"></canvas>
      <div id="unity-loading-bar">
        <div id="console-loader">
          <pre id="boot-logo"> ____ _____ _____ _     _        _    ____    ____  ____  ___ _____ _____ 
/ ___|_   _| ____| |   | |      / \  |  _ \  |  _ \|  _ \|_ _|  ___|_   _|

|____/ |_| |_____|_____|_____/_/   \_\_| \_\ |____/|_| \_\___|_|     |_|  </pre>
          <div id="console-lines" aria-live="polite"></div>
          <div id="console-progress-text">[  0%] A start job is running for Unity WebGL runtime</div>
          <div class="console-prompt"><span id="console-status">booting Stellar Drift</span><span class="console-cursor"></span></div>
        </div>
      </div>
      <div id="unity-warning"> </div>
    </div>
    <script>
      window.addEventListener("load", function () {
        if ("serviceWorker" in navigator) {
          navigator.serviceWorker.register("ServiceWorker.js");
        }
      });

      var container = document.querySelector("#unity-container");
      var canvas = document.querySelector("#unity-canvas");
      var loadingBar = document.querySelector("#unity-loading-bar");
      var consoleProgressText = document.querySelector("#console-progress-text");
      var consoleLines = document.querySelector("#console-lines");
      var consoleStatus = document.querySelector("#console-status");
      var warningBanner = document.querySelector("#unity-warning");
      var unityGame = null;
      var bootSteps = [
        { at: 0, text: "Reached target Basic System." },
        { at: 0.08, text: "Started Create Static Device Nodes in /dev." },
        { at: 0.18, text: "Started Load Unity WebGL loader." },
        { at: 0.32, text: "Started Mount compressed framework image." },
        { at: 0.48, text: "Started Compile WebAssembly module." },
        { at: 0.64, text: "Started Load game data archive." },
        { at: 0.78, text: "Started Prepare streaming assets." },
        { at: 0.92, text: "Reached target Game Runtime." },
        { at: 1, text: "Started Stellar Drift." }
      ];
      var printedBootSteps = 0;

      function blockRightClick(event) {
        if (event.type === "contextmenu" || event.button === 2) {
          event.preventDefault();
        }
      }

      function focusCanvas() {
        canvas.focus();
      }

      function handleRightMouseDown(event) {
        if (event.button !== 2) return;

        event.preventDefault();
        focusCanvas();
        if (unityGame) unityGame.SendMessage("InputManager", "PressFireFromWeb");
      }

      function handleRightMouseUp(event) {
        if (event.button !== 2) return;

        event.preventDefault();
        focusCanvas();
        if (unityGame) unityGame.SendMessage("InputManager", "ReleaseFireFromWeb");
      }

      window.addEventListener("contextmenu", blockRightClick, true);
      window.addEventListener("mouseup", handleRightMouseUp, true);
      canvas.addEventListener("contextmenu", blockRightClick, true);
      canvas.addEventListener("pointerdown", focusCanvas, true);
      canvas.addEventListener("mousedown", focusCanvas, true);
      canvas.addEventListener("mousedown", handleRightMouseDown, true);
      canvas.addEventListener("mouseup", handleRightMouseUp, true);

      function appendConsoleLine(text) {
        var line = document.createElement("div");
        line.className = "console-line";
        var status = document.createElement("span");
        status.className = "boot-ok";
        status.textContent = "[  OK  ]";
        var message = document.createElement("span");
        message.textContent = " " + text;
        line.appendChild(status);
        line.appendChild(message);
        consoleLines.appendChild(line);
        consoleLines.scrollTop = consoleLines.scrollHeight;
      }

      function updateLoadingConsole(progress) {
        var percent = Math.min(100, Math.floor(progress * 100));
        consoleProgressText.textContent = "[" + String(percent).padStart(3, " ") + "%] A start job is running for Unity WebGL runtime";

        while (printedBootSteps < bootSteps.length && progress >= bootSteps[printedBootSteps].at) {
          appendConsoleLine(bootSteps[printedBootSteps].text);
          printedBootSteps++;
        }

        if (progress < 0.35) consoleStatus.textContent = "loading game data";
        else if (progress < 0.7) consoleStatus.textContent = "starting wasm runtime";
        else if (progress < 1) consoleStatus.textContent = "waiting for first frame";
        else consoleStatus.textContent = "starting game";
      }

      // Shows a temporary message banner/ribbon for a few seconds, or
      // a permanent error message on top of the canvas if type=='error'.
      // If type=='warning', a yellow highlight color is used.
      // Modify or remove this function to customize the visually presented
      // way that non-critical warnings and error messages are presented to the
      // user.
      function unityShowBanner(msg, type) {
        function updateBannerVisibility() {
          warningBanner.style.display = warningBanner.children.length ? 'block' : 'none';
        }
        var div = document.createElement('div');
        div.innerHTML = msg;
        warningBanner.appendChild(div);
        if (type == 'error') div.style = 'background: red; padding: 10px;';
        else {
          if (type == 'warning') div.style = 'background: yellow; padding: 10px;';
          setTimeout(function() {
            warningBanner.removeChild(div);
            updateBannerVisibility();
          }, 5000);
        }
        updateBannerVisibility();
      }

      var version = "{{{ VERSION }}}"; // C#에서 치환될 부분
      var buildUrl = "Build";
      var loaderUrl = buildUrl + "/{{{ LOADER_FILENAME }}}?v=" + version;
      var config = {
        arguments: [],
        dataUrl: buildUrl + "/{{{ DATA_FILENAME }}}?v=" + version,
        frameworkUrl: buildUrl + "/{{{ FRAMEWORK_FILENAME }}}?v=" + version,
#if USE_THREADS
        workerUrl: buildUrl + "/{{{ WORKER_FILENAME }}}?v=" + version,
#endif
#if USE_WASM
        codeUrl: buildUrl + "/{{{ CODE_FILENAME }}}?v=" + version,
#endif
#if SYMBOLS_FILENAME
        symbolsUrl: buildUrl + "/{{{ SYMBOLS_FILENAME }}}?v=" + version,
#endif
        streamingAssetsUrl: "StreamingAssets",
        companyName: {{{ JSON.stringify(COMPANY_NAME) }}},
        productName: {{{ JSON.stringify(PRODUCT_NAME) }}},
        productVersion: {{{ JSON.stringify(PRODUCT_VERSION) }}},
        showBanner: unityShowBanner,
      };

      // By default Unity keeps WebGL canvas render target size matched with
      // the DOM size of the canvas element (scaled by window.devicePixelRatio)
      // Set this to false if you want to decouple this synchronization from
      // happening inside the engine, and you would instead like to size up
      // the canvas DOM size and WebGL render target sizes yourself.
      // config.matchWebGLToCanvasSize = false;

      if (/iPhone|iPad|iPod|Android/i.test(navigator.userAgent)) {
        // Mobile device style: fill the whole browser client area with the game canvas:
        var meta = document.createElement('meta');
        meta.name = 'viewport';
        meta.content = 'width=device-width, height=device-height, initial-scale=1.0, user-scalable=no, shrink-to-fit=yes';
        document.getElementsByTagName('head')[0].appendChild(meta);
      }

#if BACKGROUND_FILENAME
      canvas.style.background = "url('" + buildUrl + "/{{{ BACKGROUND_FILENAME.replace(/'/g, '%27') }}}') center / cover";
#endif
      loadingBar.style.display = "flex";
      updateLoadingConsole(0);

      var script = document.createElement("script");
      script.src = loaderUrl;
      script.onload = () => {
        createUnityInstance(canvas, config, (progress) => {
          updateLoadingConsole(progress);
        }).then((unityInstance) => {
          unityGame = unityInstance;
          updateLoadingConsole(1);
          loadingBar.style.display = "none";
        }).catch((message) => {
          appendConsoleLine("ERROR: " + message);
          alert(message);
        });
      };
      document.body.appendChild(script);
    </script>
